void setup() {
  Serial.begin(115200);
  Serial.println();

  // Setup pins
  pinMode(LED_PIN, OUTPUT);
  digitalWrite(LED_PIN, LOW);

  pinMode(WATER_PUMP_PIN, OUTPUT);
  digitalWrite(WATER_PUMP_PIN, LOW);

  pinMode(RELAY2_PIN, OUTPUT);
  digitalWrite(RELAY2_PIN, LOW);

  // Flash LED 3 times to indicate boot
  for(int i = 0; i < 3; i++) {
    digitalWrite(LED_PIN, HIGH);
    delay(100);
    digitalWrite(LED_PIN, LOW);
    delay(100);
  }

  // Reset camera via PWDN pin if available
  if (PWDN_GPIO_NUM != -1) {
    pinMode(PWDN_GPIO_NUM, OUTPUT);
    digitalWrite(PWDN_GPIO_NUM, LOW);
    delay(10);
    digitalWrite(PWDN_GPIO_NUM, HIGH);
    delay(100);
  }

  // Initialize camera
  setupCamera();

  // WiFiManager for SmartConfig
  WiFiManager wifiManager;
  wifiManager.autoConnect("ESP32-CAM");  // Creates hotspot if no WiFi credentials
  Serial.println("WiFi connected via SmartConfig");
  Serial.print("IP address: ");
  Serial.println(WiFi.localIP());

  // Connect to Blynk
  connectToBlynk();

  // Setup Blynk timers
  blynkConnectionTimer.setInterval(30000L, checkBlynkConnection);
  blynkConnectionTimer.setInterval(10000L, sendHeartbeat);
}

void setupCamera() {
  Serial.println("Initializing camera...");
  
  // Add a small delay before initializing camera
  delay(100);
  
  camera_config_t config;
  config.ledc_channel = LEDC_CHANNEL_0;
  config.ledc_timer = LEDC_TIMER_0;
  config.pin_d0 = Y2_GPIO_NUM;
  config.pin_d1 = Y3_GPIO_NUM;
  config.pin_d2 = Y4_GPIO_NUM;
  config.pin_d3 = Y5_GPIO_NUM;
  config.pin_d4 = Y6_GPIO_NUM;
  config.pin_d5 = Y7_GPIO_NUM;
  config.pin_d6 = Y8_GPIO_NUM;
  config.pin_d7 = Y9_GPIO_NUM;
  config.pin_xclk = XCLK_GPIO_NUM;
  config.pin_pclk = PCLK_GPIO_NUM;
  config.pin_vsync = VSYNC_GPIO_NUM;
  config.pin_href = HREF_GPIO_NUM;
  config.pin_sscb_sda = SIOD_GPIO_NUM;
  config.pin_sscb_scl = SIOC_GPIO_NUM;
  config.pin_pwdn = PWDN_GPIO_NUM;
  config.pin_reset = RESET_GPIO_NUM;
  
  // Lower clock speed for better stability
  config.xclk_freq_hz = 8000000;  // Reduced from 10MHz to 8MHz
  config.pixel_format = PIXFORMAT_JPEG;

  if(psramFound()){
    Serial.println("PSRAM found");
    config.frame_size = FRAMESIZE_VGA;
    config.jpeg_quality = 12;
    config.fb_count = 2;
  } else {
    Serial.println("No PSRAM detected");
    config.frame_size = FRAMESIZE_QVGA;
    config.jpeg_quality = 20;
    config.fb_count = 1;
  }

  // Try to initialize the camera - retry up to 3 times
  esp_err_t err = ESP_FAIL;
  int retry_count = 0;
  while (err != ESP_OK && retry_count < 3) {
    err = esp_camera_init(&config);
    if (err != ESP_OK) {
      Serial.printf("Camera init failed with error 0x%x, retrying... (%d/3)\n", err, retry_count+1);
      // Power cycle the camera by toggling PWDN pin if available
      if (PWDN_GPIO_NUM != -1) {
        digitalWrite(PWDN_GPIO_NUM, LOW);
        delay(100);
        digitalWrite(PWDN_GPIO_NUM, HIGH);
        delay(100);
      }
      retry_count++;
      delay(500); // Add a delay before retrying
    }
  }

  if (err != ESP_OK) {
    Serial.printf("Camera init failed with error 0x%x after multiple attempts, restarting...\n", err);
    delay(1000);
    ESP.restart();
  } else {
    Serial.println("Camera initialized successfully");
    
    // Set initial camera parameters for better results
    sensor_t * s = esp_camera_sensor_get();
    if (s) {
      // Initial sensor settings
      s->set_brightness(s, 0);     // -2 to 2
      s->set_contrast(s, 0);       // -2 to 2
      s->set_saturation(s, 0);     // -2 to 2
      s->set_special_effect(s, 0); // 0 to 6 (0 - No Effect, 1 - Negative, 2 - Grayscale, 3 - Red Tint, 4 - Green Tint, 5 - Blue Tint, 6 - Sepia)
      s->set_whitebal(s, 1);       // 0 = disable , 1 = enable
      s->set_awb_gain(s, 1);       // 0 = disable , 1 = enable
      s->set_wb_mode(s, 0);        // 0 to 4 - if awb_gain enabled (0 - Auto, 1 - Sunny, 2 - Cloudy, 3 - Office, 4 - Home)
      s->set_exposure_ctrl(s, 1);  // 0 = disable , 1 = enable
      s->set_aec2(s, 1);           // 0 = disable , 1 = enable
      s->set_gain_ctrl(s, 1);      // 0 = disable , 1 = enable
      s->set_agc_gain(s, 0);       // 0 to 30
      s->set_gainceiling(s, (gainceiling_t)0);  // 0 to 6
      s->set_bpc(s, 0);            // 0 = disable , 1 = enable
      s->set_wpc(s, 1);            // 0 = disable , 1 = enable
      s->set_raw_gma(s, 1);        // 0 = disable , 1 = enable
      s->set_lenc(s, 1);           // 0 = disable , 1 = enable
      s->set_hmirror(s, 0);        // 0 = disable , 1 = enable
      s->set_vflip(s, 0);          // 0 = disable , 1 = enable
      s->set_dcw(s, 1);            // 0 = disable , 1 = enable
    }
  }
}

void connectToBlynk() {
  Serial.println("Connecting to Blynk...");
  Blynk.config(BLYNK_AUTH_TOKEN);
  if (Blynk.connect(30000)) {
    Serial.println("Connected to Blynk server");
    for (int i = 0; i < 3; i++) {
      digitalWrite(LED_PIN, HIGH);
      delay(300);
      digitalWrite(LED_PIN, LOW);
      delay(300);
    }
  } else {
    Serial.println("Failed to connect to Blynk");
  }
}

void checkBlynkConnection() {
  if (!Blynk.connected()) {
    Serial.println("Blynk disconnected. Reconnecting...");
    Blynk.disconnect();
    Blynk.connect(15000);
    if (Blynk.connected()) Serial.println("Reconnected to Blynk");
    else Serial.println("Failed to reconnect to Blynk");
  }
}

  void sendHeartbeat() {
    if (Blynk.connected()) {
      Blynk.virtualWrite(V3, millis() / 1000);
      digitalWrite(LED_PIN, HIGH);
      delay(50);
      digitalWrite(LED_PIN, LOW);
      Serial.println("Heartbeat sent to Blynk");
    }
  }

void uploadPhoto() {
  Serial.println("Taking photo and uploading to Google Drive...");
  digitalWrite(LED_PIN, HIGH);
  Blynk.virtualWrite(V1, "Starting photo capture...");
  
  camera_fb_t * fb = esp_camera_fb_get();
  if (!fb) {
    Serial.println("Camera capture failed. Restarting...");
    Blynk.virtualWrite(V1, "Camera capture failed!");
    digitalWrite(LED_PIN, LOW);
    delay(1000);
    ESP.restart();
    return;
  }

  Serial.println("Photo captured: " + String(fb->len) + " bytes");
  Blynk.virtualWrite(V1, "Photo captured: " + String(fb->len / 1024) + " KB. Uploading...");

  const char* host = "script.google.com";
  client.setInsecure();
  if (client.connect(host, 443)) {
    String url = "/macros/s/" + myDeploymentID + "/exec?folder=" + myMainFolderName;
    client.println("POST " + url + " HTTP/1.1");
    client.println("Host: " + String(host));
    client.println("Transfer-Encoding: chunked");
    client.println();
    
    int fbLen = fb->len;
    char *input = (char *)fb->buf;
    int chunkSize = 3 * 1000;
    int chunkBase64Size = CustomBase64::enc_len(chunkSize);
    char output[chunkBase64Size + 1];

    for (int i = 0; i < fbLen; i += chunkSize) {
      if (i % (chunkSize * 5) == 0) {
        Blynk.run();
        blynkConnectionTimer.run();
      }
      int l = CustomBase64::encode(output, input, min(fbLen-i, chunkSize));
      client.print(l, HEX);
      client.print("\r\n");
      client.print(output);
      client.print("\r\n");
      input += chunkSize;
      delay(10);
      Serial.print(".");
    }

    client.print("0\r\n\r\n");
    esp_camera_fb_return(fb);

    long int StartTime = millis();
    while (!client.available()) {
      Serial.print(".");
      delay(100);
      Blynk.run();
      blynkConnectionTimer.run();
      if ((StartTime + 10000) < millis()) break;
    }

    Serial.println();
    while (client.available()) Serial.print(char(client.read()));

    Serial.println("\nUpload complete");
    Blynk.virtualWrite(V1, "Upload complete - Success!");
  } else {
    Serial.println("Connection to Google Scripts failed.");
    Blynk.virtualWrite(V1, "Upload failed - Connection error");
    esp_camera_fb_return(fb);
  }

  client.stop();
  digitalWrite(LED_PIN, LOW);
  Serial.println("-----------");
}

void loop() {
  Blynk.run();
  blynkConnectionTimer.run();
  
  unsigned long currentMillis = millis();
  if (currentMillis - previousMillis >= Interval) {
    previousMillis = currentMillis;
    if (Blynk.connected()) uploadPhoto();
    else {
      Serial.println("Skipping photo upload - Blynk not connected");
      checkBlynkConnection();
      if (Blynk.connected()) Blynk.virtualWrite(V1, "Upload skipped - Was disconnected");
    }
  }
}

BLYNK_CONNECTED() {
  Serial.println("Blynk connected event");
  Blynk.virtualWrite(V0, "ESP32-CAM Online");
  Blynk.virtualWrite(V1, "Ready for capture");
  Blynk.virtualWrite(WATER_PUMP_VPIN, digitalRead(WATER_PUMP_PIN));
  Blynk.virtualWrite(RELAY2_VPIN, digitalRead(RELAY2_PIN));

  for (int i = 0; i < 2; i++) {
    digitalWrite(LED_PIN, HIGH);
    delay(200);
    digitalWrite(LED_PIN, LOW);
    delay(200);
  }
}
